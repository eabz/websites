        function showPage(pageId) {
            // 1. Ocultar TODAS las secciones primero
            // Esto garantiza que no convivan dos secciones al mismo tiempo
            const sections = document.querySelectorAll('.page-section');
            sections.forEach(section => {
                section.classList.remove('active');
                // Forzamos la ocultaci√≥n del estilo inline si existiera
                section.style.display = 'none';
            });

            // 2. Identificar la secci√≥n objetivo
            const activeSection = document.getElementById(pageId);
            
            // 3. Mostrar la secci√≥n objetivo seg√∫n su tipo
            // A√±adimos la clase active
            activeSection.classList.add('active');

            if(pageId === 'inicio') {
                // Inicio necesita display flex para centrar el contenido verticalmente
                activeSection.style.display = 'flex';
            } else {
                // Las dem√°s p√°ginas usan block normal
                activeSection.style.display = 'block';
            }

            // 4. Cerrar men√∫ m√≥vil si est√° abierto
