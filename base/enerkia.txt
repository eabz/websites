        function showPage(pageId) {
            // Reset visual
            const sections = document.querySelectorAll('.page-section');
            sections.forEach(s => s.classList.remove('active'));
            
            // Ocultar display none despu√©s de animacion (simulado)
            sections.forEach(s => s.style.display = 'none');

            // Activar nueva secci√≥n
            const active = document.getElementById(pageId);
            
            if(pageId === 'inicio') active.style.display = 'flex';
            else active.style.display = 'block';

            // Peque√±o timeout para permitir que el navegador procese el display:block antes de a√±adir la clase de animaci√≥n
            setTimeout(() => {
                active.classList.add('active');
            }, 10);

            document.getElementById('nav-list').classList.remove('show');
            window.scrollTo(0,0);
        }

        function toggleMenu() {
            document.getElementById('nav-list').classList.toggle('show');
        }
